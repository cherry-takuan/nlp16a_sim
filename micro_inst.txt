;ADD命令
#0x0A
MOV MEM->IR2
INC IP->IP,ACC
;3ワードモード
;ピリオドから始まるサフィックスは条件実行。ここでは3ワード命令時に実行。
MOV.3wd MEM->IR3
INC.3wd IP->IP,ACC
;実行
MOV RA3->ACC
;プレフィックスがF_だとフラグ出力モード
F_ADD RA2->FLAG
ADD RA2->RA1

;SUB命令
#0x09
MOV MEM->IR2
INC IP->IP,ACC
;3ワードモード
;ピリオドから始まるサフィックスは条件実行。ここでは3ワード命令時に実行。
MOV.3wd MEM->IR3
INC.3wd IP->IP,ACC
;実行
MOV RA3->ACC
;プレフィックスがF_だとフラグ出力モード
F_SUB RA2->FLAG
SUB RA2->RA1

;MOV命令
#0x00
MOV MEM->IR2
INC IP->IP,ADDR
;3ワードモード
;ピリオドから始まるサフィックスは条件実行。ここでは3ワード命令時に実行。
INC.3wd IP->IP,ADDR
MOV.3wd MEM->IR3
;実行
MOV RA3->ACC
;プレフィックスがF_だとフラグ出力モード
F_MOV RA2->FLAG
MOV RA2->RA1
